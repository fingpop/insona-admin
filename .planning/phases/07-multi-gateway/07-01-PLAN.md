<objective>
将单网关架构升级为多网关架构，支持 10-20 个网关同时连接。核心改动：

1. **GatewayService 去单例化** — 从 `export const gatewayService = new GatewayService()` 改为可实例化的类，由 MultiGatewayService 管理
2. **新增 MultiGatewayService** — 单例管理器，维护 Map<gatewayId, GatewayService>，负责事件路由、广播
3. **数据库改造** — Gateway 表支持多记录，Device 表新增 gatewayId
4. **API 路由改造** — 所有 /api/gateway/* 路由支持 gatewayId 参数
5. **新增网关管理 API** — /api/gateway/add, /api/gateway/remove, /api/gateway/list
6. **设置页 UI** — 网关列表卡片，每张卡片显示状态、IP/端口、连接/断开/删除/诊断
7. **SSE 事件改造** — 事件携带 gatewayId 字段，前端透明消费

阶段目标：连接优先 — 实现多网关连接管理，设备按 gatewayId 自动归属，不改动现有控制流程。
</objective>

```typescript
import { PrismaClient } from "@prisma/client";
import { randomUUID } from "crypto";

const prisma = new PrismaClient();

async function main() {
  // 1. 先给 Device 表添加 gatewayId（通过 Prisma migrate 已完成 schema 变更）
  // 2. 获取旧的 default 网关
  const oldGw = await prisma.gateway.findUnique({ where: { id: "default" } });
  if (!oldGw) { console.log("No default gateway, skipping migration"); return; }

  // 3. 创建新网关记录
  const newId = randomUUID();
  await prisma.gateway.create({
    data: { id: newId, ip: oldGw.ip, port: oldGw.port, status: oldGw.status, lastSeen: oldGw.lastSeen }

  // 4. 更新所有现有设备的 gatewayId
  await prisma.device.updateMany({ where: {}, data: { gatewayId: newId } });

  // 5. 删除旧的 default 记录
  await prisma.gateway.delete({ where: { id: "default" } });

  console.log(`Migrated default gateway to ${newId}`);
}

main().then(() => prisma.$disconnect()).catch(console.error);
```

```typescript
import { GatewayService } from "./GatewayService";
import { prisma } from "@/lib/prisma";

type SSEConsumer = (data: string) => void;

class MultiGatewayService {
  private gateways: Map<string, GatewayService> = new Map();
  private sseConsumers: Set<SSEConsumer> = new Set();

  // 获取指定网关实例
  getGateway(gatewayId: string): GatewayService | undefined {
    return this.gateways.get(gatewayId);
  }

  // 获取所有已连接的网关
  getConnectedGateways(): GatewayService[] {
    return Array.from(this.gateways.values()).filter(gw => gw.isConnected);
  }

  // 连接网关（从数据库读取配置后调用）
  async connectGateway(gatewayId: string, ip: string, port: number): Promise<void> {
    if (this.gateways.has(gatewayId)) {
      throw new Error(`Gateway ${gatewayId} already exists`);
    }

    const gw = new GatewayService(gatewayId);

    // 订阅该网关的 SSE 事件并广播
    gw.subscribeSSE((data) => {
      // 在事件中加入 gatewayId
      const enriched = this._enrichEvent(data, gatewayId);
      this._broadcast(enriched);
    });

    this.gateways.set(gatewayId, gw);
    await gw.connect(ip, port);
  }

  // 断开并移除网关
  async removeGateway(gatewayId: string): Promise<void> {
    const gw = this.gateways.get(gatewayId);
    if (gw) {
      await gw.disconnect();
      this.gateways.delete(gatewayId);
    }
  }

  // 断开指定网关
  async disconnectGateway(gatewayId: string): Promise<void> {
    const gw = this.gateways.get(gatewayId);
    if (gw) {
      await gw.disconnect();
    }
  }

  // 连接指定网关（不移除实例）
  async connectStoredGateway(gatewayId: string): Promise<void> {
    const gw = this.gateways.get(gatewayId);
    if (gw) {
      await gw.connect(gw.ip, gw.port); // 需要 GatewayService 暴露 ip/port getter
    }
  }

  // SSE 订阅
  subscribeSSE(onData: SSEConsumer): () => void {
    this.sseConsumers.add(onData);
    return () => this.sseConsumers.delete(onData);
  }

  // 从数据库加载并连接所有网关
  async loadAndConnectAll(): Promise<void> {
    const gateways = await prisma.gateway.findMany({
      where: { status: { not: "disconnected" } } // 或者始终尝试连接所有
    });

    for (const gw of gateways) {
      try {
        await this.connectGateway(gw.id, gw.ip, gw.port);
      } catch (err) {
        console.error(`[MultiGateway] Failed to connect ${gw.ip}:${gw.port}:`, err);
        await prisma.gateway.update({
          where: { id: gw.id },
          data: { status: "error" }
        });
      }
    }
  }

  private _enrichEvent(data: string, gatewayId: string): string {
    try {
      const parsed = JSON.parse(data);
      return JSON.stringify({ ...parsed, gatewayId });
    } catch {
      // 非 JSON 数据也包装为事件格式
      return JSON.stringify({ type: "raw", data, gatewayId });
    }
  }

  private _broadcast(data: string) {
    for (const consumer of this.sseConsumers) {
      try { consumer(data); } catch { /* ignore */ }
    }
  }
}

export const multiGatewayService = new MultiGatewayService();
```
